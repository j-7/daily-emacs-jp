#+BEGIN: include :file "/r/sync/junk/151121/151121063432.ibuffer-tramp.el"

  (lambda ()
    (ibuffer-tramp-set-filter-groups-by-tramp-connection)
    (ibuffer-do-sort-by-alphabetic)))
